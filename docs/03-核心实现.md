### 节点提取器

```typescript
// services/nodeExtractor.ts

interface ExtractorOptions {
  includeHidden: boolean;
  includeDescendants: boolean;
  maxDepth: number;
}

export async function extractNodes(
  nodes: SceneNode[],
  options: ExtractorOptions = {
    includeHidden: false,
    includeDescendants: true,
    maxDepth: 10,
  }
): Promise<ExtractedNode[]> {
  const result: ExtractedNode[] = [];

  for (const node of nodes) {
    const extracted = await extractNode(node, options, 0);
    if (extracted) {
      result.push(extracted);
    }
  }

  return result;
}

async function extractNode(
  node: SceneNode,
  options: ExtractorOptions,
  depth: number
): Promise<ExtractedNode | null> {
  // 跳过隐藏节点
  if (!options.includeHidden && node.visible === false) {
    return null;
  }

  // 超过最大深度
  if (depth >= options.maxDepth) {
    return null;
  }

  // 基础信息
  const baseNode: ExtractedNode = {
    id: node.id,
    name: node.name,
    type: node.type,
    absoluteBoundingBox: node.absoluteBoundingBox,
  };

  // 提取类型特定属性
  if ('fills' in node) {
    baseNode.fills = extractFills(node.fills);
  }

  if ('strokes' in node) {
    baseNode.strokes = extractStrokes(node.strokes);
  }

  if ('effects' in node) {
    baseNode.effects = extractEffects(node.effects);
  }

  if ('cornerRadius' in node) {
    baseNode.cornerRadius = node.cornerRadius;
  }

  if ('opacity' in node) {
    baseNode.opacity = node.opacity;
  }

  // 处理组件属性
  if (node.type === 'COMPONENT' || node.type === 'COMPONENT_SET') {
    baseNode.componentProperties = extractComponentProperties(
      node as ComponentNode
    );
  }

  // 处理变体
  if (node.type === 'COMPONENT_SET') {
    baseNode.variants = extractVariants(node as ComponentSetNode);
  }

  // 递归处理子节点
  if (options.includeDescendants && 'children' in node) {
    baseNode.children = [];
    for (const child of node.children) {
      const childNode = await extractNode(child, options, depth + 1);
      if (childNode) {
        baseNode.children.push(childNode);
      }
    }
  }

  return baseNode;
}
```

```typescript
// services/styleExtractor.ts

export function extractFills(fills: readonly Paint[] | figma.mixed): Fill[] {
  if (!fills || fills === figma.mixed) return [];

  return fills.map((fill) => {
    if (fill.type === 'SOLID') {
      return {
        type: 'solid',
        color: {
          r: Math.round(fill.color.r * 255),
          g: Math.round(fill.color.g * 255),
          b: Math.round(fill.color.b * 255),
          a: fill.opacity ?? 1,
        },
        opacity: fill.opacity,
      };
    }

    if (fill.type === 'GRADIENT_LINEAR') {
      return {
        type: 'gradient',
        gradient: extractGradientStops(fill.gradientStops),
        angle: fill.gradientAngle,
      };
    }

    return { type: 'unsupported', fillType: fill.type };
  });
}

export function extractEffects(effects: readonly Effect[]): ExtractedEffect[] {
  if (!effects) return [];

  return effects.map((effect) => {
    switch (effect.type) {
      case 'DROP_SHADOW':
        return {
          type: 'dropShadow',
          color: {
            r: Math.round(effect.color.r * 255),
            g: Math.round(effect.color.g * 255),
            b: Math.round(effect.color.b * 255),
            a: effect.color.a,
          },
          offset: effect.offset,
          blur: effect.blurRadius,
          spread: effect.spread,
        };

      case 'INNER_SHADOW':
        return {
          type: 'innerShadow',
          color: {
            r: Math.round(effect.color.r * 255),
            g: Math.round(effect.color.g * 255),
            b: Math.round(effect.color.b * 255),
            a: effect.color.a,
          },
          offset: effect.offset,
          blur: effect.blurRadius,
        };

      case 'BLUR':
        return {
          type: 'blur',
          blur: effect.radius,
        };

      default:
        return { type: 'unsupported', effectType: effect.type };
    }
  });
}
```

```typescript
// core/src/generators/component-generator.ts

import { Project, SourceFile, SyntaxKind } from 'ts-morph';

export class ComponentGenerator {
  private project: Project;

  constructor(options: GeneratorOptions) {
    this.project = new Project({
      manipulationSettings: {
        indentationText: 2,
        newLineKind: 'lf',
      },
    });
  }

  generate(design: DesignNode): GeneratedComponent {
    const sourceFile = this.project.createSourceFile(
      `${design.name}.tsx`,
      '',
      { overwrite: true }
    );

    // 添加导入
    this.addImports(sourceFile);

    // 生成接口
    const interfaceDeclaration = this.addInterface(
      sourceFile,
      design.name,
      design.props
    );

    // 生成组件
    const componentDeclaration = this.addComponent(
      sourceFile,
      design.name,
      interfaceDeclaration.getName()
    );

    // 生成导出
    this.addExports(sourceFile, design.name);

    return {
      name: design.name,
      code: sourceFile.getFullText(),
      language: 'tsx',
      props: design.props,
    };
  }

  private addImports(sourceFile: SourceFile): void {
    sourceFile.addImportDeclaration({
      namedImports: ['FC'],
      moduleSpecifier: 'react',
    });

    if (this.options.includeCSS) {
      sourceFile.addImportDeclaration({
        defaultImport: `'./${this.options.componentName}.css'`,
        moduleSpecifier: `./${this.componentName}.css`,
      });
    }
  }

  private addInterface(
    sourceFile: SourceFile,
    name: string,
    props: PropDefinition[]
  ): InterfaceDeclarationStructure {
    return sourceFile.addInterface({
      name: `${name}Props`,
      isExported: true,
      properties: props.map((prop) => ({
        name: prop.required ? prop.name : `${prop.name}?` as string,
        type: prop.type,
      })),
    });
  }

  private addComponent(
    sourceFile: SourceFile,
    name: string,
    interfaceName: string
  ): ClassDeclarationStructure {
    return sourceFile.addFunction({
      name,
      isExported: true,
      statements: (writer) => {
        writer.writeLine(`const ${name}: FC<${interfaceName}> = (`);
        writer.writeLine('  props,');
        writer.writeLine(') => {');
        writer.indent(() => {
          writer.writeLine('const {');
          writer.writeLine('  children,');
          writer.writeLine('  ...rest');
          writer.writeLine('} = props;');
          writer.writeLine('');
          writer.writeLine('return (');
          writer.writeLine('  <div');
          writer.writeLine('    className={styles.container}');
          writer.writeLine('    {...rest}');
          writer.writeLine('  >');
          writer.writeLine('    {children}');
          writer.writeLine('  </div>');
          writer.writeLine(');');
        });
        writer.writeLine('};');
      },
    });
  }
}
```

### Props 推断器

```typescript
// core/src/inference/props-inferrer.ts

export class PropsInferrer {
  infer(design: DesignNode): PropDefinition[] {
    const props: PropDefinition[] = [];

    // 1. 从 Component Properties 推断
    if (design.componentProperties) {
      const componentProps = this.inferFromComponentProperties(
        design.componentProperties
      );
      props.push(...componentProps);
    }

    // 2. 从 Variant 推断
    if (design.type === 'COMPONENT_SET') {
      const variantProps = this.inferFromVariants(design.variants);
      props.push(...variantProps);
    }

    // 3. 从命名推断
    const namedProps = this.inferFromNaming(design.name);
    props.push(...namedProps);

    // 4. 从交互状态推断
    const interactionProps = this.inferFromInteractionStates(design);
    props.push(...interactionProps);

    return this.deduplicateAndSort(props);
  }

  private inferFromComponentProperties(
    properties: Record<string, ComponentProperty>
  ): PropDefinition[] {
    return Object.entries(properties).map(([key, value]) => {
      switch (value.type) {
        case 'BOOLEAN':
          return {
            name: this.camelCase(key),
            type: 'boolean',
            required: false,
            defaultValue: value.defaultValue ?? false,
            description: `From Figma component property: ${key}`,
          };

        case 'TEXT':
          return {
            name: this.camelCase(key),
            type: 'string',
            required: false,
            defaultValue: value.defaultValue ?? '',
            description: `From Figma component property: ${key}`,
          };

        case 'VARIANT':
          return {
            name: 'variant',
            type: this.generateVariantType(value.variantOptions ?? []),
            required: true,
            description: 'Component variant',
            enum: value.variantOptions,
          };

        default:
          return {
            name: this.camelCase(key),
            type: 'unknown',
            required: false,
          };
      }
    });
  }

  private inferFromNaming(name: string): PropDefinition[] {
    const props: PropDefinition[] = [];
    const lowerName = name.toLowerCase();

    // disabled 推断
    if (lowerName.includes('disabled')) {
      props.push({
        name: 'disabled',
        type: 'boolean',
        required: false,
        defaultValue: false,
        description: 'Disabled state',
      });
    }

    // loading 推断
    if (lowerName.includes('loading')) {
      props.push({
        name: 'loading',
        type: 'boolean',
        required: false,
        defaultValue: false,
        description: 'Loading state',
      });
    }

    // size 推断
    if (lowerName.includes('size') || lowerName.includes('small')) {
      props.push({
        name: 'size',
        type: "'small' | 'medium' | 'large'",
        required: false,
        defaultValue: 'medium',
        description: 'Component size',
        enum: ['small', 'medium', 'large'],
      });
    }

    return props;
  }

  private inferFromInteractionStates(design: DesignNode): PropDefinition[] {
    const props: PropDefinition[] = [];

    // 检查是否有 hover 状态图层
    const hasHoverLayer = design.children?.some(
      (child) => child.name.toLowerCase().includes('hover')
    );
    if (hasHoverLayer) {
      props.push({
        name: 'hovered',
        type: 'boolean',
        required: false,
        defaultValue: false,
        description: 'Hover state',
      });
    }

    // 检查是否有 focus 状态图层
    const hasFocusLayer = design.children?.some(
      (child) => child.name.toLowerCase().includes('focus')
    );
    if (hasFocusLayer) {
      props.push({
        name: 'focused',
        type: 'boolean',
        required: false,
        defaultValue: false,
        description: 'Focus state',
      });
    }

    return props;
  }

  private camelCase(str: string): string {
    return str
      .replace(/[-_\s]+(.)?/g, (_, char) => (char ? char.toUpperCase() : ''))
      .replace(/^(.)/, (char) => char.toLowerCase());
  }
}
```

```typescript
// core/src/generators/story-generator.ts

export class StoryGenerator {
  generate(component: GeneratedComponent, design: DesignNode): StoryFile {
    const stories: Story[] = [];

    // Default story
    stories.push(this.generateDefaultStory(component));

    // Variant stories
    if (design.variants) {
      stories.push(...this.generateVariantStories(component, design.variants));
    }

    // Interaction state stories
    stories.push(...this.generateInteractionStories(component, design));

    return {
      title: this.generateTitle(design),
      component: component.name,
      meta: this.generateMeta(component, design),
      stories,
    };
  }

  private generateMeta(
    component: GeneratedComponent,
    design: DesignNode
  ): StoryMeta {
    return {
      title: this.generateTitle(design),
      component: component.name,
      parameters: {
        layout: 'centered',
        docs: {
          description: {
            component: `Auto-generated from Figma: ${design.name}`,
          },
        },
      },
      argTypes: this.generateArgTypes(component.props),
    };
  }

  private generateArgTypes(props: PropDefinition[]): ArgTypeDefinition[] {
    return props.map((prop) => ({
      name: prop.name,
      type: { name: prop.type },
      defaultValue: prop.defaultValue,
      description: prop.description,
      control: this.inferControlType(prop),
      table: {
        type: { summary: prop.type },
        defaultValue: { summary: String(prop.defaultValue) },
      },
    }));
  }

  private inferControlType(prop: PropDefinition): ControlConfig {
    if (prop.enum && prop.enum.length <= 5) {
      return { type: 'select', options: prop.enum };
    }

    switch (prop.type) {
      case 'boolean':
        return { type: 'boolean' };
      case 'string':
        return { type: 'text' };
      case 'number':
        return { type: 'number' };
      default:
        return { type: 'object' };
    }
  }

  private generateTitle(design: DesignNode): string {
    // 将组件名转换为 Storybook title 格式
    // 例如: "Button Primary" -> "Components/Button/Primary"
    const parts = design.name.split(' ');
    if (parts.length === 1) {
      return `Components/${parts[0]}`;
    }

    // 假设最后一个部分是变体
    const variant = parts.pop();
    return `Components/${parts.join('/')}/${variant}`;
  }
}
```

## 3.4 样式转换器

```typescript
// core/src/converters/style-converter.ts

export class StyleConverter {
  convert(design: DesignNode, options: StyleOptions): StyleOutput {
    const styles: Record<string, string> = {};

    // 基础样式
    if (design.absoluteBoundingBox) {
      styles.width = `${design.absoluteBoundingBox.width}px`;
      styles.height = `${design.absoluteBoundingBox.height}px`;
    }

    // 背景色
    if (design.fills?.length) {
      styles.backgroundColor = this.convertFills(design.fills);
    }

    // 边框
    if (design.strokes?.length) {
      styles.border = this.convertStrokes(design.strokes);
    }

    // 圆角
    if (design.cornerRadius) {
      styles.borderRadius = `${design.cornerRadius}px`;
    }

    // 阴影
    if (design.effects?.length) {
      styles.boxShadow = this.convertEffects(design.effects);
    }

    // 透明度
    if (design.opacity !== undefined && design.opacity !== 1) {
      styles.opacity = String(design.opacity);
    }

    return {
      css: this.stylesToCSS(design.name, styles),
      tailwind: options.format === 'tailwind' ? this.toTailwind(styles) : null,
      tokens: this.extractTokens(design),
    };
  }

  private convertFills(fills: Fill[]): string {
    const solid = fills.find((f) => f.type === 'solid');
    if (solid && solid.type === 'solid') {
      return this.colorToRGBA(solid.color);
    }

    const gradient = fills.find((f) => f.type === 'gradient');
    if (gradient && gradient.type === 'gradient') {
      return this.gradientToCSS(gradient.gradient);
    }

    return 'transparent';
  }

  private convertEffects(effects: ExtractedEffect[]): string {
    const shadows = effects.filter(
      (e) => e.type === 'dropShadow' || e.type === 'innerShadow'
    );

    return shadows
      .map((shadow) => {
        if (shadow.type === 'dropShadow') {
          return `${shadow.offset.x}px ${shadow.offset.y}px ${shadow.blur}px rgba(${shadow.color.r}, ${shadow.color.g}, ${shadow.color.b}, ${shadow.color.a})`;
        }
        if (shadow.type === 'innerShadow') {
          return `inset ${shadow.offset.x}px ${shadow.offset.y}px ${shadow.blur}px rgba(${shadow.color.r}, ${shadow.color.g}, ${shadow.color.b}, ${shadow.color.a})`;
        }
        return 'none';
      })
      .join(', ');
  }

  private colorToRGBA(color: Color): string {
    if (color.a !== undefined && color.a < 1) {
      return `rgba(${color.r}, ${color.g}, ${color.b}, ${color.a})`;
    }
    return `rgb(${color.r}, ${color.g}, ${color.b})`;
  }

  private stylesToCSS(className: string, styles: Record<string, string>): string {
    const selector = `.${this.toKebabCase(className)}`;
    const properties = Object.entries(styles)
      .map(([key, value]) => `  ${this.toKebabCase(key)}: ${value};`)
      .join('\n');

    return `${selector} {\n${properties}\n}`;
  }
}
```
